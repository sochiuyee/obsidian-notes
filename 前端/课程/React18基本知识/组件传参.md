原生的event对象在nativeEvent上
```js
class Button extends React.Component {
      state = {
        count: 1
      }
      handleClick = (e) => {
        console.log('e---', e)
      }

      render() {
        return (
          <div>
            <button type="button" onClick={this.handleClick}>Click Me!</button>
            <div>{this.state.count}</div>
          </div>
        )
      }
    }
```

###### 注意this指向问题，推荐使用剪头函数
```js
class Button extends React.Component {
      handleClick = (e) => {
	      console.log('this---', this) // <Button 对象，在点击时调用了handleClick，此时调用对象是button
       }

      handleClick() {
        console.log('this---', this) // undefined，此时调用的对象是
        // class 严格模式下，this指向undefined
      }

      render() {
        return (
          <div>
            <button type="button" onClick={this.handleClick}>Click Me!</button>
            <div>{this.state.count}</div>
          </div>
        )
      }
    }
```

###### 接收参数
```js
	handleClick = (e, data) => {
        console.log('e----', e)
        console.log('data----', data)
      }

      render() {
        return (
          <div>
          // 如果是要接收event和data参数，需要用箭头函数传参e
            <button type="button" onClick={(e) => this.handleClick(e, 123)}>Click Me!</button>
            <div>{this.state.count}</div>
          </div>
        )
      }


	handleClick = (data) => {
        console.log('data----', data) // 只能接收传递的参数
      }

      render() {
        return (
          <div>
          // 如果是要接收event和data参数，需要用箭头函数传参e
            <button type="button" onClick={this.handleClick(e, 123)}>Click Me!</button>
            <div>{this.state.count}</div>
          </div>
        )
      }
```


==事件传参推荐使用高阶函数==
```js
	handleClick = num => {
        return ev => {
          console.log('ev---', ev)
          console.log('num----', num)
        }
      }

      render() {
        return (
          <div>
            <button type="button" onClick={this.handleClick(123)}>Click Me!</button>
            <div>{this.state.count}</div>
          </div>
        )
      }
```
