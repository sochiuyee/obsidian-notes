> [!TIP] 非受控组件将数据存储在`DOM`中，而不是组件内，这比较类似于传统的`HTML`表单元素。
> 1.  非受控组件的值`不受`组件自身的`state`和`props`控制
> 2.  非受控组件使用`ref`从`DOM`中获取元素数据


```js
<body>
  <div id="root"></div>
  <script type="text/babel">
    const domNode = document.getElementById('root');
    class Button extends React.Component {
      constructor(props) {
        super(props)
        this.ref = React.createRef()
      }
      
      state = {
        count: 1

      }

      changeState = () => {
		// 获取到组件的dom实例
        const content = this.ref.current
        this.setState({
          count: content.value
        })
      }

      render() {
        return (
          <div>
            <input type="text" ref={this.ref}/> // 通过ref去获取组件的dom
            {this.state.count}
            <button type="button" onClick={this.changeState}>click</button>
          </div>
        )
      }
    }


    let element = (
      <Button />
    )
    
    ReactDOM.render(element, domNode)
  </script>
</body>
```
